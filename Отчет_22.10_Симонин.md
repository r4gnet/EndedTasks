d=open('enc.bin','rb').read()
for k in range(1,256):
dec = bytes([b ^ k for b in d])
  
if b'FLAG{' in dec:
  
    try:
      
        s = dec.decode('utf-8', 
          
      errors='ignore')
    except:
      
        s = dec.decode('latin1', errors='ignore')

    print(f"FOUND KEY=0x{k:02X}
      
      FLAG={m.group(0) if m else s}")
    break
